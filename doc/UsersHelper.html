    Return a string containing the first user role (if present)

    # File app/helpers/users_helper.rb, line 6
def get_user_role(user)
  roles = user.roles.collect{|r| r.name}
  if roles.empty?
    return "(none)"      
  else
    return roles[0]
  end
end
